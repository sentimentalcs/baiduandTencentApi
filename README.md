### 2.linux操作
      cd /usr/local
      tar zxvf xpdfbin-linux-3.04.tar.gz -C /usr/local
       
   　　`cd /usr/local/xpdfbin-linux-3.04`  
   
   　　`cat INSTALL`
   
   　　`cd bin32/`
   
   　　`cp ./* /usr/local/bin/`
   
   　　`cd ../doc/`
   
   　　`mkdir -p /usr/local/man/man1`
   
   　　`mkdir -p /usr/local/man/man5`
   
   　　`cp *.1 /usr/local/man/man1`
   
   　　`cp *.5 /usr/local/man/man5<br/>`
